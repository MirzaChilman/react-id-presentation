import * as React from "react";
import { bindActionCreators, Action } from "redux";

import {
  AkDynamicTableStateless as DynamicTable,
  AkDropdownMenu as DropdownMenu,
  AkDropdownItemGroup as DropdownItemGroup,
  AkDropdownItem as DropdownItem,
  Card,
  Tooltip,
  PermissionsCheckList,
} from "~/components/common-components";
import { BackupStore, Store, StoresState, storeActionType } from "@packages/model/stores";
import * as StoreActions from "@packages/stores/actions/StoreActions";
import { translation } from "~/i18n";
import CreateStoreModal from "../CreateStoreModal";
import history from "~/utils/history";
import DeleteStoreModal from "../DeleteStoreModal";
import EditStoreModal from "../EditStoreModal";
import CloneStoreModal from "../CloneStoreModal";
import ImportStoreModal from "./StoreActionModal/ImportStoreModal";
import PublishStoreModal from "../PublishStoreModal";
import { StoreSchema } from "~/models/Stores";
import { formatDateForDetailedDisplay } from "~/components/common-components/utils/utils";
import RollbackStoreModal from "@packages/stores/Action/Rollback";

interface Props {
  stores: StoreSchema[];
  StoreActions: typeof StoreActions;
  backupStore: BackupStore;
}
interface State {
  isCreateStoreModalOpen: boolean;
  isDeleteStoreModalOpen: boolean;
  isEditStoreModalOpen: boolean;
  isCloneStoreModalOpen: boolean;
  isImportStoreModalOpen: boolean;
  isPublishStoreModalOpen: boolean;
  isRollbackStoreModalOpen: boolean;
  oneStore: StoreSchema;
  isLoadingBackup: boolean;
  isRollingBack: boolean;
}

class StoresPage extends React.Component<Props, State> {
   constructor(props: Props) {
     super(props);
     this.state = {
       isCreateStoreModalOpen: false,
       isDeleteStoreModalOpen: false,
       isEditStoreModalOpen: false,
       isCloneStoreModalOpen: false,
       isImportStoreModalOpen: false,
       isPublishStoreModalOpen: false,
       isRollbackStoreModalOpen: false,
       oneStore: new Store(),
       isLoadingBackup: false,
       isRollingBack: false,
     };
   }

   render() {
       const { stores } = this.props;
       const {
         storePermissions: { adminCanReadStore },
       } = PermissionsCheckList;
       return (
         <div className="page-content-wrap">
           <Card
             noPadding={true}
             noMarginHorizontal={true}
             cardTitle={translation("stores.published")}
             isAuthorized={adminCanReadStore()}
           >
             <div className="styled-atlaskit-table">
               {!stores.filter((store: StoreSchema) => store.published).length ? (
                 <div className="empty-list-wrap">
                   <p>{translation("stores.empty")}</p>
                 </div>
               ) : (
                 <DynamicTable head={this.getTableHead()} rows={this.getTableRows({ published: true })} />
               )}
             </div>
           </Card>
           {this.renderDraftStore(stores)}
           {this.renderCreateStoreModal()}
           {this.renderDeleteStoreModal()}
           {this.renderEditStoreModal()}
           {this.renderCloneStoreModal()}
           {this.renderPublishStoreModal()}
           {this.renderImportStoreModal()}
           {this.renderRollbackStoreModal()}
         </div>
       );
     }
}
